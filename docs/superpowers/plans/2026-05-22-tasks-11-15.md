```python
"""配置中心"""
from __future__ import annotations
from pathlib import Path
from typing import Any
import yaml

class ConfigCenter:
    def __init__(self, config_path: Path):
        self._config_path = Path(config_path)
        self._data: dict[str, Any] = {}
        self._load()
    def _load(self) -> None:
        if self._config_path.exists():
            with open(self._config_path, encoding="utf-8") as f: self._data = yaml.safe_load(f) or {}
    def get(self, key: str, default: Any = None) -> Any: return self._data.get(key, default)
    def set(self, key: str, value: Any) -> None: self._data[key] = value
    def save(self) -> None:
        with open(self._config_path, "w", encoding="utf-8") as f: yaml.dump(self._data, f, allow_unicode=True)
```

```python
from __future__ import annotations
from typing import Any
from uuid import uuid4

class Session:
    def __init__(self, session_id: str):
        self.session_id = session_id
        self.context: dict[str, Any] = {}
    def set(self, key: str, value: Any) -> None: self.context[key] = value
    def get(self, key: str, default: Any = None) -> Any: return self.context.get(key, default)

class SessionManager:
    def __init__(self): self._sessions: dict[str, Session] = {}
    def create(self) -> Session:
        sid = str(uuid4()); s = Session(sid); self._sessions[sid] = s; return s
    def get(self, session_id: str) -> Session | None: return self._sessions.get(session_id)
```

```python
import pytest
import yaml
from pathlib import Path
from core.kernel import Kernel

class TestKernel:
    def _make_config(self, tmp_path: Path) -> Path:
        cfg = tmp_path / "config.yaml"
        cfg.write_text(yaml.dump({"app": {"name": "test"}}))
        return cfg

    def _make_plugin(self, tmp_path: Path, name: str = "p1") -> Path:
        plugins_dir = tmp_path / "plugins"
        d = plugins_dir / name
        d.mkdir(parents=True)
        (d / "plugin.yaml").write_text(yaml.dump({"name": name, "version": "0.1.0", "type": "tool"}))
        return plugins_dir

    def test_init(self, tmp_path):
        cfg = self._make_config(tmp_path)
        plugins = self._make_plugin(tmp_path)
        policies = tmp_path / "policies"
        policies.mkdir()
        k = Kernel(cfg, plugins, policies)
        assert k.config is not None
        assert k.event_bus is not None
        assert k.plugin_registry is not None
        assert k.session_manager is not None

    def test_config_get_set(self, tmp_path):
        cfg = self._make_config(tmp_path)
        plugins = tmp_path / "plugins"; plugins.mkdir()
        policies = tmp_path / "policies"; policies.mkdir()
        k = Kernel(cfg, plugins, policies)
        assert k.config.get("app") == {"name": "test"}
        k.config.set("debug", True)
        assert k.config.get("debug") is True

    def test_plugin_registry_discovered(self, tmp_path):
        cfg = self._make_config(tmp_path)
        plugins = self._make_plugin(tmp_path, "my-plugin")
        policies = tmp_path / "policies"; policies.mkdir()
        k = Kernel(cfg, plugins, policies)
        assert len(k.plugin_registry.list_plugins()) == 1
        assert k.plugin_registry.get_meta("my-plugin") is not None

    def test_event_bus_pub_sub(self, tmp_path):
        cfg = self._make_config(tmp_path)
        plugins = tmp_path / "plugins"; plugins.mkdir()
        policies = tmp_path / "policies"; policies.mkdir()
        k = Kernel(cfg, plugins, policies)
        received = []
        k.event_bus.subscribe("test", lambda e: received.append(e))
        k.event_bus.publish("test", {"x": 1})
        assert len(received) == 1 and received[0]["x"] == 1
```

```python
"""安装脚本"""
from __future__ import annotations
import platform
import subprocess
import sys
from pathlib import Path
import yaml

REQUIRED_PYTHON = (3, 10)

def detect_platform() -> str:
    system = platform.system().lower()
    if system == "windows": return "windows"
    if system == "darwin": return "macos"
    return "linux"

def check_python() -> bool:
    v = sys.version_info
    return v.major >= REQUIRED_PYTHON[0] and v.minor >= REQUIRED_PYTHON[1]

def init_config(config_path: Path) -> None:
    default = {"app": {"name": "supermedicine", "version": "0.1.0"}, "plugins": {"dir": "plugins"}, "policies": {"dir": "policies"}}
    config_path.parent.mkdir(parents=True, exist_ok=True)
    with open(config_path, "w", encoding="utf-8") as f:
        yaml.dump(default, f, allow_unicode=True)

def main() -> None:
    print(f"[SuperMedicine] Platform: {detect_platform()}")
    if not check_python():
        print(f"[ERROR] Python >= {REQUIRED_PYTHON[0]}.{REQUIRED_PYTHON[1]} required, got {sys.version}")
        sys.exit(1)
    print("[SuperMedicine] Python version OK")
    config_path = Path("config.yaml")
    init_config(config_path)
    print(f"[SuperMedicine] Config written to {config_path}")
    print("[SuperMedicine] Installation complete.")

if __name__ == "__main__":
    main()
```

```python
"""RAG Provider 接口"""
from __future__ import annotations
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any

@dataclass
class RAGResult:
    content: str
    score: float
    metadata: dict[str, Any]

class RAGProvider(ABC):
    @abstractmethod
    def query(self, text: str, top_k: int = 5) -> list[RAGResult]:
        """检索相关文档"""
        ...
    @abstractmethod
    def index(self, documents: list[dict[str, Any]]) -> None:
        """索引文档"""
        ...
    @abstractmethod
    def health_check(self) -> bool:
        """健康检查"""
        ...

class EmptyRAGProvider(RAGProvider):
    def query(self, text: str, top_k: int = 5) -> list[RAGResult]:
        return []
    def index(self, documents: list[dict[str, Any]]) -> None:
        pass
    def health_check(self) -> bool:
        return True
```

```python
"""Agent 监控"""
from __future__ import annotations
import json
from pathlib import Path
from typing import Any

class AgentMonitor:
    def __init__(self, audit_log: Path):
        self._audit_log = Path(audit_log)
    def get_permission_audit(self, agent_id: str | None = None, limit: int = 100) -> list[dict[str, Any]]:
        """获取权限审计记录"""
        if not self._audit_log.exists(): return []
        entries = []
        with open(self._audit_log, encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line: continue
                entry = json.loads(line)
                if agent_id is None or entry.get("agent_id") == agent_id:
                    entries.append(entry)
        return entries[-limit:]
    def get_denied_actions(self, agent_id: str | None = None) -> list[dict[str, Any]]:
        """获取被拒绝的操作"""
        all_entries = self.get_permission_audit(agent_id, limit=0)
        return [e for e in all_entries if e.get("result") == "DENIED"]
    def detect_anomalies(self, agent_id: str | None = None, threshold: int = 5) -> list[dict[str, Any]]:
        """检测异常行为（短时间内大量拒绝）"""
        denied = self.get_denied_actions(agent_id)
        if len(denied) >= threshold:
            return denied
        return []
```
